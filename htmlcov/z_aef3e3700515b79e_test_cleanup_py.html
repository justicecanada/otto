32def test_redundant_librarian_upload(client, all_apps_user): 

33 """ 

34 Upload a file through Librarian modal. 

35 Check media directory 

36 Upload same file again. 

37 Check media directory - still should be just 1 instance of the file though there are now 2 documents 

38 Delete one of the documents in Librarian modal. 

39 File should still be there, since there is a reference from the other Document object. 

40 Delete the other Document 

41 File should be gone now. 

42 """ 

43 user = all_apps_user() 

44 client.force_login(user) 

45 # Create a DataSource in the user library (manually) 

46 user_library = user.personal_library 

47 data_source = DataSource.objects.create( 

48 library=user_library, 

49 name="Test Data Source", 

50 ) 

51 upload_url = reverse("librarian:upload", kwargs={"data_source_id": data_source.id}) 

52 # Test uploading this file 

53 this_file_path = os.path.abspath(__file__) 

54 with open(this_file_path, "rb") as f: 

55 response = client.post(upload_url, {"file": f}) 

56 assert response.status_code == 200 

57 # Check SavedFile created 

58 saved_files = SavedFile.objects.all() 

59 assert saved_files.count() == 1 

60 # Check Document created 

61 documents = data_source.documents.all() 

62 assert documents.count() == 1 

63 # Check media directory 

64 assert os.path.exists(saved_files[0].file.path) 

65 # Upload the same file again 

66 with open(this_file_path, "rb") as f: 

67 response = client.post(upload_url, {"file": f}) 

68 assert response.status_code == 200 

69 # Check SavedFile NOT created 

70 saved_files = SavedFile.objects.all() 

71 assert saved_files.count() == 1 

72 # A document should NOT have been created since hash, filename and data source are the same 

73 documents = data_source.documents.all() 

74 assert documents.count() == 1 

75 # Now, change the filename and upload again 

76 # First, copy this file to a different name 

77 new_file_path = os.path.join(os.path.dirname(this_file_path), "new_file_name.txt") 

78 shutil.copy(this_file_path, new_file_path) 

79 with open(new_file_path, "rb") as f: 

80 response = client.post(upload_url, {"file": f}) 

81 assert response.status_code == 200 

82 # Delete the file 

83 os.remove(new_file_path) 

84 # Check SavedFile NOT created 

85 saved_files = SavedFile.objects.all() 

86 assert saved_files.count() == 1 

87 # A document should have been created since the filename is different 

88 documents = data_source.documents.all() 

89 assert documents.count() == 2 

90 # Check that both documents reference the same SavedFile 

91 assert documents.first().file == documents.last().file 

92 # Check media directory. There should be only 1 file 

93 folder = os.path.dirname(saved_files[0].file.path) 

94 assert len(os.listdir(folder)) == 1 

95 # Delete one of the documents 

96 document = documents.first() 

97 document.delete() 

98 # Check SavedFile still exists 

99 saved_files = SavedFile.objects.all() 

100 assert saved_files.count() == 1 

101 # Check media directory 

102 assert os.path.exists(saved_files[0].file.path) 

103 # Delete the other document 

104 document = documents.last() 

105 document.delete() 

106 # Check SavedFile is gone 

107 saved_files = SavedFile.objects.all() 

108 assert saved_files.count() == 0 

109 # Check media directory 

110 len(os.listdir(folder)) == 0

114def test_redundant_chat_upload(client, all_apps_user): 

115 """ 

116 Start a new chat. 

117 Upload a file using chunk_upload route. 

118 Check the media directory for the file 

119 Upload the same file again. 

120 Check the media directory - there should still be only 1 file. 

121 Both files should be in the library. 

122 Delete the chat 

123 Check the media directory - files should be gone. 

124 """ 

125 user = all_apps_user() 

126 client.force_login(user) 

127 # Start a new chat 

128 response = client.get(reverse("chat:new_chat")) 

129 assert response.status_code == 302 

130 # Check that the chat was created 

131 chat = user.chat_set.first() 

132 assert chat is not None 

133 

134 # Create a message in the chat 

135 message = Message.objects.create(chat=chat) 

136 # Upload a file 

137 this_file_path = os.path.abspath(__file__) 

138 # Hash of this file 

139 with open(this_file_path, "rb") as f: 

140 sha256_hash = generate_hash(f.read()) 

141 # Try uploading the file through the chat "chunk upload" mechanism 

142 # We don't need to worry about the actual chunking; just upload the whole thing. 

143 upload_url = reverse("chat:chunk_upload", kwargs={"message_id": message.id}) 

144 with open(this_file_path, "rb") as f: 

145 response = client.post( 

146 upload_url, 

147 { 

148 "file": f, 

149 "hash": sha256_hash, 

150 "filename": os.path.basename(this_file_path), 

151 "end": 1, 

152 "file_id": "null", 

153 "nextSlice": "null", 

154 "content_type": "text/plain", 

155 }, 

156 ) 

157 assert response.status_code == 200 

158 

159 # Check that a SavedFile was created 

160 saved_files = SavedFile.objects.all() 

161 assert saved_files.count() == 1 

162 # Check that a ChatFile was created 

163 chat_files = ChatFile.objects.all() 

164 assert chat_files.count() == 1 

165 # Check media directory 

166 assert os.path.exists(saved_files[0].file.path) 

167 

168 # Upload again, this time indicating the upload is not complete (end=0) 

169 # but since the hash is included, it should be recognized as a duplicate 

170 upload_url = reverse("chat:chunk_upload", kwargs={"message_id": message.id}) 

171 with open(this_file_path, "rb") as f: 

172 response = client.post( 

173 upload_url, 

174 { 

175 "file": f, 

176 "hash": sha256_hash, 

177 "filename": os.path.basename(this_file_path), 

178 "end": 0, 

179 "file_id": "null", 

180 "nextSlice": 123, 

181 "content_type": "text/plain", 

182 }, 

183 ) 

184 assert response.status_code == 200 

185 # Check there is still only one SavedFile 

186 saved_files = SavedFile.objects.all() 

187 assert saved_files.count() == 1 

188 # Check that there are now two ChatFiles 

189 chat_files = ChatFile.objects.all() 

190 assert chat_files.count() == 2 

191 # Check that both ChatFiles references the SavedFile 

192 assert chat_files.first().saved_file == saved_files.first() 

193 # Check media directory 

194 assert os.path.exists(saved_files[0].file.path) 

195 # Make sure there is still only one file 

196 folder = os.path.dirname(saved_files[0].file.path) 

197 assert len(os.listdir(folder)) == 1 

198 # Delete the chat 

199 chat.delete() 

200 # Check that the SavedFile is gone 

201 saved_files = SavedFile.objects.all() 

202 assert saved_files.count() == 0 

203 # Check media directory 

204 assert len(os.listdir(folder)) == 0
